# Phase 1: Write the defineService Call

Define the service with its identity, command, readiness detection, and lifecycle config.

## Steps

### 1. Set identity fields

```typescript
import { defineService } from '../framework.js';

export const myService = defineService({
  id: 'my-service',          // Kebab-case — used for command names and state files
  name: 'My Service',        // Display name for logs and status output
  description: 'Runs the development server',
});
```

### 2. Set the command

**Static command** — always the same:
```typescript
command: 'pnpm dev',
```

**Parameterized command** — varies per invocation:
```typescript
command: (params) => {
  const parts = ['pnpm', 'dev'];
  if (params?.port) parts.push('-p', String(params.port));
  if (params?.watch !== false) parts.push('--watch');
  return parts.join(' ');
},
```

### 3. Add paramsSchema (if needed)

```typescript
import { z } from 'zod';

const myServiceParams = z.object({
  port: z.coerce.number().default(3000).describe('Dev server port'),
  watch: z.boolean().default(true).describe('Enable file watching'),
});

export const myService = defineService({
  // ...
  paramsSchema: myServiceParams,
});
```

Use `z.coerce.number()` for port parameters — CLI flags are always strings.

### 4. Add environment variables

```typescript
env: (config, params) => ({
  PORT: String(params?.port ?? 3000),
  NODE_ENV: 'development',
  HOST: '0.0.0.0',
}),
```

The `env` function receives the resolved CLI config and start params. Return a flat `Record<string, string>`.

### 5. Configure readiness detection

**Single pattern** — one line to match:
```typescript
readiness: {
  pattern: /Local:\s*http:\/\/localhost:(\d+)/,
  captures: { 'server-url': 1 },
  timeout: 30_000,
},
```

**Multiple patterns** — all must match:
```typescript
readiness: {
  patterns: [
    {
      name: 'web-ui',
      pattern: /Local:\s*(http:\/\/localhost:\d+)/,
      captures: { 'web-url': { group: 1, type: 'website' } },
    },
    {
      name: 'api',
      pattern: /API server:\s*(http:\/\/[^\s]+)/,
      captures: { 'api-url': { group: 1, type: 'api-rest' } },
    },
    {
      name: 'mcp',
      pattern: /MCP server available at (http:\/\/[^\s]+)/,
      captures: { 'mcp-url': { group: 1, type: 'api-mcp' } },
    },
  ],
  timeout: 60_000,
},
```

Tips:
- Use `[^\s]+` to match URLs (stops at whitespace)
- Set timeout high enough for slow services (Next.js builds can take 60s+)
- The `name` on each pattern appears in `service:pattern-matched` events for progress feedback
- Use typed captures (`{ group, type }`) for endpoints that should be discoverable

### 6. Configure shutdown

```typescript
shutdown: { signal: 'SIGTERM', timeout: 10_000 },
```

The framework sends `signal` first. If the process doesn't exit within `timeout` ms, SIGKILL is sent.

### 7. Configure restart (optional)

```typescript
restart: { enabled: true, maxRetries: 3, delay: 5_000 },
```

On crash, the service auto-restarts up to `maxRetries` times with `delay` ms between attempts.

### 8. Set maxInstances (if needed)

```typescript
maxInstances: 1,   // Default — exclusive service (e.g., dev server)
maxInstances: 4,   // Parallel workers
```

### 9. Assemble the full definition

```typescript
import { defineService, checkWorkdir, checkCommand, checkPort } from '../framework.js';
import { z } from 'zod';

const devServerParams = z.object({
  port: z.coerce.number().default(3000).describe('Dev server port'),
});

export const devServer = defineService({
  id: 'dev-server',
  name: 'Dev Server',
  description: 'Next.js development server',
  command: (params) => `pnpm dev -p ${params?.port ?? 3000}`,
  paramsSchema: devServerParams,
  env: (_config, params) => ({
    PORT: String(params?.port ?? 3000),
    NODE_ENV: 'development',
  }),
  preflight: [
    checkWorkdir(
      (cwd) => fs.existsSync(path.join(cwd, 'package.json')),
      'Not a Node.js project',
      'Run dev-server-start --workdir <project-dir>',
    ),
    checkPort((ctx) => (ctx.params?.port as number) ?? 3000, 'Dev Server'),
  ],
  readiness: {
    pattern: /Local:\s*http:\/\/localhost:(\d+)/,
    captures: { 'server-url': 1 },
    timeout: 60_000,
  },
  shutdown: { signal: 'SIGTERM', timeout: 10_000 },
  restart: { enabled: true, maxRetries: 3, delay: 5_000 },
});
```
